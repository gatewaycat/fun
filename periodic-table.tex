\newcounter{atom}

\newcommand\period[3]{
  \setcounter{atom}{#2}
  \foreach \n in {#3}{
    \addtocounter{atom}{-1}
    \draw({(\arabic{atom}-#2)*\x}, #1*-\x)
      node 
        [label=above:{\scalebox{5}{\arabic{atom}}}, outer sep=1mm]
        {\scalebox{10}{\smash\n\vphantom0}};
  }
}
