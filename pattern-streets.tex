\RequirePackage{luatex85}
\documentclass[tikz]{standalone}
\usepackage{pgfmath}

\def\x{3.000mm} \def\xskip{0.5mm} \def\xsep{16.25mm}
\def\y{3.375mm} \def\yskip{2.25mm} \def\ysep{9mm}

\newcommand\block[2]{
	\def\a{\ifnum#1<10 0#1\else#1\fi}
	\def\b{\ifnum#2<10 0#2\else#2\fi}
	\pgfmathtruncatemacro\rawc{#1+2}
	\pgfmathtruncatemacro\rawd{#2+2}
	\def\c{\ifnum\rawc<10 0\rawc\else\rawc\fi}
	\def\d{\ifnum\rawd<10 0\rawd\else\rawd\fi}
	\draw (#1*\xsep,#2*\ysep)
	node [anchor=east] {{\bfseries\a}\b9}
	++(down:\y) node [anchor=east] {{\bfseries\a}\b7}
	++(down:\y) node [anchor=east] {{\bfseries\a}\b5}
	++(down:\y) node [anchor=east] {{\bfseries\a}\b3}
	++(down:\y) node [anchor=east] {{\bfseries\a}\b1}
	++(\xskip,-\yskip) node [anchor=west, rotate=90] {{\bfseries\b}\a1}
	++(right:\x) node [anchor=west, rotate=90] {{\bfseries\b}\a3}
	++(right:\x) node [anchor=west, rotate=90] {{\bfseries\b}\a5}
	++(right:\x) node [anchor=west, rotate=90] {{\bfseries\b}\a7}
	++(right:\x) node [anchor=west, rotate=90] {{\bfseries\b}\a9}
	++(\xskip,\yskip) node [anchor=east, rotate=180] {{\bfseries\c}\b0}
	++(up:\y) node [anchor=east, rotate=180] {{\bfseries\c}\b2}
	++(up:\y) node [anchor=east, rotate=180] {{\bfseries\c}\b4}
	++(up:\y) node [anchor=east, rotate=180] {{\bfseries\c}\b6}
	++(up:\y) node [anchor=east, rotate=180] {{\bfseries\c}\b8}
	++(-\xskip,\yskip) node [anchor=west, rotate=270] {{\bfseries\d}\a8}
	++(left:\x) node [anchor=west, rotate=270] {{\bfseries\d}\a6}
	++(left:\x) node [anchor=west, rotate=270] {{\bfseries\d}\a4}
	++(left:\x) node [anchor=west, rotate=270] {{\bfseries\d}\a2}
	++(left:\x) node [anchor=west, rotate=270] {{\bfseries\d}\a0}
	;
	}
