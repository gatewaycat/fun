\begin{tikzpicture}[ultra thick, scale=0.25]
\pgfmathtruncatemacro{\nmax}{\numberofcolumns*\numberofrows-1}

\draw ({mod(\n,\numberofcolumns)*3.95cm},{-floor(\n/\numberofcolumns)*3.48cm}) coordinate (temp);

\pgfmathtruncatemacro{\temp}{mod(floor(mod(\n,64)/2^\e),2)}
	\draw (temp) ++(up:0.25) coordinate (temp);
	\ifnum\temp=0 \draw (temp) -- ++(right:1) ++(right:1) -- ++(right:1);
	\else \draw (temp) -- ++(right:3); \fi
	\draw (temp) ++(up:0.25) coordinate (temp);
	}
}
