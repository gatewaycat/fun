		(0,2) ..controls +(90:1.5cm) and +(90:1.5cm) ..

		(2,2)  .. controls +(90:1.5cm) and  +(90:1.5cm) ..

		;
	\draw (0,-4)

		edge[-] ++(right:1.9)
		;
	}}

\newcounter{card}
\newcommand\card[1]{
%% switch black/red
\color{\suitcolor}
%% page break every 9 cards
\ifthenelse{\equal{\arabic{card}}{9}}{\newpage\setcounter{card}0}{}
\begin{tikzpicture}
		[remember picture, overlay]
	%% position 3x3 cards to a page
	\draw (current page.north west)
		++({(mod(\arabic{card},3)+0.5)*1/3*8.5in},
			{(floor(\arabic{card}/3)+0.5)*-1/3*11in})
		coordinate (center);
	%% position content on given card
	\foreach \content/\x/\y/\scale/\angle in {#1}{
		\edef\temp{\noexpand\draw (center) ++(\x*1/3*8.5in,\y*1/3*11in)
		node[rotate=\angle] {\noexpand\scalebox{\scale}{$\content$}};}
		\temp
		}
	%% cut guides
	\foreach \x/\y in {1/1, 1/-1, -1/1, -1/-1}
	\draw[gray] (center) ++(\x*1/2*1/3*8.5in, \y*1/2*1/3*11in)
		edge[-] ++(left:8mm)
		edge[-] ++(right:8mm)
		edge[-] ++(up:8mm)
		edge[-] ++(down:8mm);
	\end{tikzpicture}
	\stepcounter{card}
	}

	\noexpand\suit/-0.35/-0.3/3/180,
