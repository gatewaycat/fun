\newcounter{card}
\newcommand\card[1]{
% page break every 9 cards
\ifthenelse{\equal{\arabic{card}}{9}}{\newpage\setcounter{card}0}{}
\begin{tikzpicture}
		[remember picture, overlay]
	% position 3x3 cards to a page
	\draw (current page.north west)
		++({(floor(\arabic{card}/3)+0.5)*1/3*8.5in},
			{(mod(\arabic{card},3)+0.5)*-1/3*11in})
		coordinate (center);
	% position content on given card
	\foreach \content/\x/\y/\scale/\angle in {#1}{
		\edef\temp{\noexpand\draw (center) ++(\x*1/3*8.5in,\y*1/3*11in) node[rotate=\angle] {\noexpand\scalebox{\scale}{$\content$}};}
		\temp
		}
	% cut guides
	\foreach \x/\y in {1/1, 1/-1, -1/1, -1/-1}
	\draw[gray] (center) ++(\x*1/2*1/3*8.5in, \y*1/2*1/3*11in)
		edge[-] ++(left:8mm)
		edge[-] ++(right:8mm)
		edge[-] ++(up:8mm)
		edge[-] ++(down:8mm);
	\end{tikzpicture}
	\stepcounter{card}
	}
\newcommand\suitwrap[1]{\begin{tikzpicture}[scale=0.06]
	#1
	\end{tikzpicture}}
\newcommand\hand{\suitwrap{
	\fill[bend left=90, rotate=180]
		(0,2) ..controls +(90:1.5cm) and +(90:1.5cm) .. 
		(-2,2) .. controls  +(-90:1cm) and +(90:1cm) .. 
		(-1,-1) to (1,-1) .. controls +(90:1cm) and +(-90:1cm) ..
		(2,2)  .. controls +(90:1.5cm) and  +(90:1.5cm) .. 
		cycle
		;
	\draw (0,-4) 
		edge[-] ++(up:4) 
		edge[-] ++(left:1.9)
		edge[-] ++(right:1.9)
		;
	}}
